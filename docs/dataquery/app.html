 14def upload_files(file_ext):
 15    """
 16    Uploads data files and zip archives.
 17
 18    Args:
 19        file_ext (list): A list of file extensions to filter the uploaded files.
 20
 21    Returns:
 22        list: A list of uploaded files stored in `st.session_state.uploaded_files`.
 23
 24    Notes:
 25        - Handles file uploads via Streamlit's file uploader.
 26        - Removes views from the database for files that are no longer uploaded.
 27    """
 28    # Upload files object
 29    uploaded_files = st.file_uploader(
 30        "Choose data files",
 31        accept_multiple_files=True,
 32        help='Upload your data files and zip archives.  \nAll files from zip archives and all sheets of xlsx files will be considered.',
 33        type=file_ext,
 34    )
 35
 36    # Check for removed files
 37    removed_files = [file for file in st.session_state.uploaded_files if file not in uploaded_files]
 38    # Remove views from the database for removed files
 39    for file in removed_files:
 40        tables_to_remove = [table for table, source in st.session_state.tables.items() if source == file.name]
 41        for table in tables_to_remove:
 42            remove_view(st.session_state.con, table)
 43            del st.session_state.tables[table]
 44        st.warning(f"Removed file: {file.name} and its associated tables")
 45
 46    # Update the list of uploaded files
 47    st.session_state.uploaded_files = uploaded_files
 48
 49    return st.session_state.uploaded_files

 72def files_to_db(file_ext):
 73    """
 74    Loads files into a DuckDB database and registers them as tables.
 75
 76    Args:
 77        file_ext (list): A list of file extensions to be loaded.
 78
 79    Returns:
 80        dict: A dictionary mapping table names to file names.
 81
 82    Notes:
 83        - Logs messages about successfully loaded tables and excluded files.
 84        - Handles zip archives and extracts supported files for loading.
 85        - Registers tables in the DuckDB connection.
 86    """
 87    loaded_tab = ""
 88    excluded_tab = ""
 89    file_options = {}
 90    new_tables_list = []
 91
 92    for file in st.session_state.uploaded_files:
 93        if file.name not in [f.name for f in st.session_state.uploaded_files if f != file]:
 94            file_extension = file.name.split('.')[-1].lower()
 95
 96            # Manage zip archives
 97            if file.type == "application/x-zip-compressed":
 98                with zipfile.ZipFile(file) as z:
 99                    for zip_info in z.infolist():
100                        if not zip_info.is_dir():
101                            _, extension = splitext(zip_info.filename)
102                            if extension.lstrip('.').lower() in file_ext:
103                                with z.open(zip_info) as zf:
104                                    extracted_file = BytesIO(zf.read())
105                                    extracted_file.name = basename(zip_info.filename)
106                                    # Get file options
107                                    options = get_file_options(
108                                        extracted_file.name,
109                                        None if extension.lstrip('.').lower() != 'xlsx' else pd.ExcelFile(extracted_file).sheet_names,
110                                        file.name,
111                                    )
112
113                                    file_options[extracted_file.name] = options
114                                    # Load tables and register views in DuckDB
115                                    loaded_tables = files_to_table(extracted_file, st.session_state.con, options, file.name)
116                                    if loaded_tables:
117                                        for table in loaded_tables:
118                                            st.session_state.tables[table] = extracted_file.name
119                                            new_tables_list.append(table)
120                                        tables_list = ''.join([f'  \n- {t}' for t in loaded_tables])
121                                        loaded_tab += f"Loaded {file.name} - {extracted_file.name} as table(s):{tables_list}  \n\n"
122                            else:
123                                excluded_tab += f"{file.name} - {zip_info.filename} not loaded. Unsupported file format  \n"
124            # Manage single files
125            else:
126                # Get file options
127                options = get_file_options(file.name, None if file_extension != 'xlsx' else pd.ExcelFile(file).sheet_names)
128
129                # Load tables and register views in DuckDB
130                file_options[file.name] = options
131                loaded_tables = files_to_table(file, st.session_state.con, options)
132                if loaded_tables:
133                    for table in loaded_tables:
134                        st.session_state.tables[table] = file.name
135                        new_tables_list.append(table)
136                    tables_list = ''.join([f'  \n- {t}' for t in loaded_tables])
137                    loaded_tab += f"Loaded {file.name} as table(s):{tables_list}  \n\n"
138
139    # Cleanup obsolete tables (renamed aliases or unselected sheets)
140    saved_tables = st.session_state.get('saved_tables', set())
141    tables_to_remove = [table for table in list(st.session_state.tables.keys())
142                        if table not in new_tables_list and table not in saved_tables]
143    for table in tables_to_remove:
144        remove_view(st.session_state.con, table)
145        del st.session_state.tables[table]
146
147    # Display success and warning messages
148    if loaded_tab != "":
149        st.success(loaded_tab)
150    if excluded_tab != "":
151        st.warning(excluded_tab)
152
153    return st.session_state.tables

237def files_to_table(file, con, options=None, archive_name=None):
238    """
239    Get dataframe from file and register it in a database connection.
240
241    Args:
242        file: File object or path to the file.
243        con: Database connection object.
244        options: Dict of options for file loading.
245        archive_name: Name of the zip archive.
246
247    Returns:
248        table_names(List): List of table names where the data was registered, or None if there was an error.
249    """
250    # Get file extension, file name, and archive name
251    file_extension = file.name.split('.')[-1].lower()
252    file_nm = f"{archive_name.replace('.', '_')}_{file.name.replace('.', '_')}" if archive_name else file.name.replace('.', '_')
253    table_names = []
254
255    try:
256        # Manage CSV and TXT using collected file settings
257        if file_extension in ['csv', 'txt']:
258            delim = '\t' if options['delimiter'] == '\\t' else options['delimiter']
259            # Read CSV using utf-8 encoding
260            df = pd.read_csv(
261                file,
262                sep=delim,
263                quoting=getattr(csv, options.get('quoting', 'QUOTE_NONE')),
264                quotechar=options.get('quotechar', '"'),
265                header=options.get('header', 0),
266                dtype=options.get('dtype', str),
267                on_bad_lines='skip',
268                encoding_errors='ignore',
269            )
270            # In case of no header, rename columns from simple integer to col_integer
271            if options.get('header', 0) is None:
272                df.columns = [f'col_{i+1}' for i in range(len(df.columns))]
273        # Manage XLSX using header settings collected
274        elif file_extension == 'xlsx':
275            xls = pd.ExcelFile(file)
276            dfs = {}
277            selected_sheets = options.get('selected_sheets', xls.sheet_names)
278            # Loop selected sheets and in case of no header, rename columns from simple integer to col_integer
279            for sheet_name in xls.sheet_names:
280                if sheet_name not in selected_sheets:
281                    continue
282                sheet_options = options.get('sheets', {}).get(sheet_name, {})
283                dfs[sheet_name] = pd.read_excel(file, sheet_name=sheet_name, header=sheet_options.get('header', 0))
284                if sheet_options.get('header', 0) is None:
285                    dfs[sheet_name].columns = [f'col_{i+1}' for i in range(len(dfs[sheet_name].columns))]
286        # Manage other accepted file formats
287        elif file_extension == 'parquet':
288            df = pd.read_parquet(file)
289        elif file_extension == 'avro':
290            df = pdx.read_avro(file, na_dtypes=True)
291        elif file_extension == 'json':
292            json_data = json.load(file)
293            if isinstance(json_data, dict):
294                # It's a single object
295                df = pd.DataFrame([json_data])
296            else:
297                # It's a list of objects
298                df = pd.json_normalize(json_data)
299        elif file_extension == 'xml':
300            df = pd.read_xml(file)
301        else:
302            st.error(f"File {file.name} not loaded. Unsupported file format.")
303            return None
304
305        # Register dataframes into DuckDB
306        if file_extension == 'xlsx':
307            used_names = {}
308            for sheet_name, df in dfs.items():
309                sheet_options = options.get('sheets', {}).get(sheet_name, {})
310                alias = sheet_options.get('alias', '').strip()
311                if alias:
312                    resolved_name = clean_table_name(alias)
313                else:
314                    resolved_name = clean_table_name(f"{file_nm}_{sheet_name.lower()}")
315                # Check for duplicate table names
316                if resolved_name in used_names:
317                    st.error(f"Duplicate alias: sheet \"{sheet_name}\" resolves to table \"{resolved_name}\", "
318                             f"already used by sheet \"{used_names[resolved_name]}\". Skipping.")
319                    continue
320                used_names[resolved_name] = sheet_name
321                table_name = register_dataframe(con, df, resolved_name)
322                table_names.append(table_name)
323        else:
324            alias = options.get('alias', '').strip() if options else ''
325            if alias:
326                resolved_name = clean_table_name(alias)
327            else:
328                resolved_name = clean_table_name(file_nm)
329                
330            table_name = register_dataframe(con, df, resolved_name)
331            table_names.append(table_name)
332
333        return table_names
334
335    except Exception as e:
336        st.error(f"Error loading file {file.name}: {str(e)}")
337        # Manage exception of wrong file settings provided for CSV and TXT
338        if file_extension in ['csv', 'txt'] and "Error tokenizing data" in str(e):
339            st.warning(f'{file.name} not loaded. Please check file settings.')
340        else:
341            st.error(f"Error loading file {file.name}: {str(e)}")
342        return None

365def clean_table_name(name):
366    """
367    Cleans the table name by replacing spaces with underscores and removing all other special characters.
368    Adds an underscore if the name starts or ends with a number.
369
370    Args:
371        name (str): The table name to be cleaned.
372
373    Returns:
374        name(str): The cleaned table name.
375    """
376    # Replace spaces with underscores
377    name = name.replace(' ', '').lower()
378    # Remove all other special characters
379    name = sub(r'[^a-zA-Z0-9_]', '', name)
380
381    return name

420def build_session_zip():
421    """
422    Builds a ZIP archive containing one parquet file per session table.
423
424    The archive is built entirely in memory (BytesIO): no data is ever written
425    to the server filesystem, respecting the application's data policy.
426    Each table is exported as "<table_name>.parquet" so that re-uploading the
427    ZIP restores the session with the same table names.
428
429    Returns:
430        bytes: The ZIP archive as bytes, or None if it could not be built.
431    """
432    con = st.session_state.con
433    zip_buffer = BytesIO()
434
435    try:
436        with zipfile.ZipFile(zip_buffer, 'w', zipfile.ZIP_DEFLATED) as zf:
437            # Write each session table as a parquet file
438            for table_name in st.session_state.tables:
439                try:
440                    df = con.execute(f'SELECT * FROM "{table_name}"').fetchdf()
441                except Exception as e:
442                    st.warning(f"Could not export table '{table_name}': {e}")
443                    continue
444                parquet_buffer = BytesIO()
445                df.to_parquet(parquet_buffer, index=False, engine='pyarrow')
446                zf.writestr(f"{table_name}.parquet", parquet_buffer.getvalue())
447    except Exception as e:
448        st.error(f"Could not build session file: {e}")
449        return None
450
451    return zip_buffer.getvalue()

725    """
